MaxDepth Of a Tree 

Algorithm
Compute the "maxDepth" of a tree -- the number of nodes along
the longest path from the root node down to the farthest leaf node.
 
In the above tree the maximum depth is 3.
 
Source Code
#include <iostream >
using namespace std;
struct node
{
    int data;
    node *left;
    node *right;
};

node *newnode(int val)
{
    node *ptr=new node();
    ptr->data=val;
    ptr->left=NULL;
    ptr->right=NULL;
    return(ptr);
}
int maxdepth (node *ptr)
{
    if(ptr==NULL)
    {
        return(0);
    }
    else
    {
        int leftsub=maxdepth(ptr->left);
        int rightsub=maxdepth(ptr->right);

        if(leftsub>=rightsub)
        {
            return(leftsub+1);
        }
        else
        {
            return(rightsub+1);
        }

    }
}

int main()
{
    node *root=newnode(1);
    root->left=newnode(2);
    root->right=newnode(3);
    root->left->left=newnode(4);
    root->left->right=newnode(5);
    cout<<"max depth is "<<maxdepth(root);
    }
