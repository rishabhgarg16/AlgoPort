Same tree or not 

Algorithm
Given two trees, return true if they are structurally identical.
 	 
are not same.
      
Source Code
#include <iostream >
using namespace std;
struct node
{
    int data;
    node *left;
    node *right;
};

node *newnode(int val)
{
    node *ptr=new node();
    ptr->data=val;
    ptr->left=NULL;
    ptr->right=NULL;
    return(ptr);
}

bool sametree(node *ptr1,node *ptr2)
{
    if(ptr1==NULL&&ptr2==NULL)
    {
        return(true);
    }
    if(ptr1!=NULL&&ptr2!=NULL)
    {
        return((ptr1->data&&ptr2->data)
               &&sametree(ptr1->left,ptr2->left)
               &&sametree(ptr1->right,ptr2->right)
              );
    }
    else
    {
        return(false);
    }
}

int main()
{
    //first tree
    node *root1=newnode(1);
    root1->left=newnode(2);
    root1->right=newnode(3);
    root1->left->left=newnode(4);
    root1->left->right=newnode(5);

    //creation of second tree
    node *root2=newnode(1);
    root2->left=newnode(2);
    root2->right=newnode(3);
    root2->left->left=newnode(4);
    //root2->left->right=newnode(5);


    if(sametree(root1,root2))
    {
        cout<<"same tree"<<endl;
    }
    else
    {
        cout<<"different"<<endl;
    }
}
