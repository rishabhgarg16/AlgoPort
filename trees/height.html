Size Of A Tree 

Algorithm
Computes the number of nodes in a tree
For example: The tree below has 5 nodes

      
Source Code
#include <iostream >
using namespace std;
struct node
{
    int data;
    node *left;
    node *right;
};

node *newnode(int val)
{
    node *ptr=new node();
    ptr->data=val;
    ptr->left=NULL;
    ptr->right=NULL;
    return(ptr);
}

int sizetree(node *ptr)
{
    if(ptr==NULL)
    {
        return(0);
    }
    else
    {
        return(sizetree(ptr->left)+1+sizetree(ptr->right));
    }
}

int main()
{
    node *root=newnode(1);
    root->left=newnode(2);
    root->right=newnode(3);
    root->left->left=newnode(4);
    root->left->right=newnode(5);
    int size;
    size=sizetree(root);
    cout <<size;
}
