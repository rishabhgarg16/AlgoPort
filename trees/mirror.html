Mirror Of a Tree

Algorithm

Change a tree so that the roles of the
left and right pointers are swapped at every node.
 
       
Source Code
#include <iostream >
using namespace std;
struct node
{
    int data;
    node *left;
    node *right;
};

node *newnode(int val)
{
    node *ptr=new node();
    ptr->data=val;
    ptr->left=NULL;
    ptr->right=NULL;
    return(ptr);
}

void mirror(node *ptr)
{
    if(ptr==NULL)
    {
        return;
    }
    else
    {
        mirror(ptr->left);
        mirror(ptr->right);

        node *temp;
        temp=ptr->left;
        ptr->left=ptr->right;
        ptr->right=temp;
    }
}

void printmirror(node *ptr)
{
    if(ptr==NULL)
    {
        return;
    }
    else
    {
        cout<<ptr->data;
        printmirror(ptr->left);
        printmirror(ptr->right);
    }
}

int main()
{
    node *root=newnode(1);
    root->left=newnode(2);
    root->right=newnode(3);
    root->left->left=newnode(4);
    root->left->right=newnode(5);
    cout<<"original tree(preorder traversal)"<<endl;
    printmirror(root);
    mirror(root);
    cout<<endl<<"after mirror(preorder traversal)"<<endl;
    printmirror(root);
}
